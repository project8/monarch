271  inline void M3Header::WriteScalarToHDF5( H5::H5Location* aLoc, const std::string& aName, const std::string& aValue )
272  {
273  M3DEBUG( mlog_mheader, "Writing string to new scalar metadata <" << aName << ">: " << aValue << "; size = " << aValue.size() );
274  // aName.c_str() and aValue.c_str() are used because while using the std::string itself, the value was getting mangled
275  H5::DataType tType = MH5Type< std::string >::H5( aValue );
276  aLoc->createAttribute( aName.c_str(), tType, H5::DataSpace( H5S_SCALAR ) ).write( tType, aValue.c_str() );
277  return;
278  }
