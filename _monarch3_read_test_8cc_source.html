303 bool ReadRecordCheck( const M3Stream* aStream, int aOffset, unsigned aDataFormat )
304 {
305  if( ! aStream->ReadRecord( aOffset ) )

310 
311  switch( aDataFormat )

320  case sDigitizedS:
321  if( ! PrintChannelsUInt( aStream ) )
322  {
323  ERROR( mlog, "Failed to print channels (digitized)" );
324  return false;
325  }
326  break;
327  case sAnalog:
328  switch( aStream->GetSampleSize() )
329  {
330  case 1:
331  if( ! PrintChannelsFloat( aStream ) )

333  ERROR( mlog, "Problem printing channels (float)" );
334  return 0;

341  return 0;

343  break;
344  }

346  default:
